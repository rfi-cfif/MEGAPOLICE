git config --list - чтобы посмотреть конфигурацию вашего git
git config --global user.name "Name"  - изменить имя пользователя
git config --global user.email "email"  - изменить email пользователя
git init - создать репозиторий
rm -rf .git - удалить репозиторий
cd <адрес этой папки> (для перехода к другой папке в командной строке)
cd /D <адрес папки> (если надо переключаться между дисками C и D)
git add <название файла> - начать остлеживать файл
git add . - добавить все файлы в отслеживаемые
git log - посмотреть все коммиты
git commit - сделать коммит изменения файла
git commit -m "..." - сделать коммит, сразу назвав его
git checkout <хеш> - переключение между коммитами
git checkout -b <название ветки> - создать ветку и переключиться в неё
git branch <название ветки> - создать ветку, но в неё не переключаться
git branch -d <название ветки> - удалить ветку
git checkout <название ветки> - переключиться в ветку
git branch - посмотреть все ветки
git log --graph --oneline --all --abbrev-commit - посмотреть дерево коммитов
